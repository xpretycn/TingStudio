# 配方列表布局改进 - 编码测试验收方案

## 一、方案概述

### 目标
将当前三段式垂直布局重构为**3:1 左右双栏布局**，优化首屏信息密度和小助手曝光率。

### 核心改进点
| 改进项 | 当前状态 | 改进后 |
|-------|---------|--------|
| 首屏可见列表行数 | 2-3行 | **8-10行** |
| 小助手曝光率 | <15% | **60%+** |
| 仪表盘数据刷新 | 静态写死 | 接入真实数据 |
| 操作列宽度 | 135px | **80px**（图标仅） |
| 次要列（负责人/更新时间） | 显示 | **隐藏→详情页** |

const columns = computed(() => {
  const screenWidth = window.innerWidth;
  
  const baseColumns = [
    { colKey: 'row-select', type: 'multiple', width: 48, resizable: false },
    { colKey: 'name', title: sortTitle('配方信息', 'name'), width: 180, ellipsis: true },

    return [
      ...baseColumns,
      { colKey: 'salesmanName', title: '负责人', width: 130, align: 'left' },
      { colKey: 'salesQuantity', title: '销量', width: 90, align: 'center' },
      { colKey: 'createdAt', title: sortTitle('更新时间', 'createdAt'), width: 140 },

    return [
      ...baseColumns,
      { colKey: 'operation', title: '操作', width: 80, align: 'center' }, // 紧凑操作
    ];
  } else {
    return baseColumns;
  }
});

## 四、Playwright 自动化测试脚本

### 测试文件：`test/formula-list-layout.spec.ts`

```typescript
import { test, expect } from '@playwright/test';
import { loginAsAdmin } from './utils/auth';
import { takeScreenshot } from './utils/screenshot';

test.describe('配方列表布局改进', () => {
  test.beforeEach(async ({ page }) => {
    await loginAsAdmin(page);
    await page.goto('/formulas');
    await page.waitForLoadState('networkidle');
  });

  test('1920px - 3:1布局，8列可见，首屏≥8行', async ({ page }) => {
    await page.setViewportSize({ width: 1920, height: 1080 });
    
    await takeScreenshot(page, 'layout-1920px');
    
    // 验证侧边栏
    await expect(page.locator('.sidebar')).toBeVisible();
    await expect(page.locator('.dashboard-mini')).toBeVisible();
    await expect(page.locator('.activity-card--assistant')).toBeVisible();
    
    // 验证列：8列（1440px显示全部）
    const visibleColumns = await page.locator('.t-table-header th').count();
    expect(visibleColumns).toBeGreaterThanOrEqual(7);
    
    // 验证首屏行数
    const visibleRows = await page.locator('.t-table-tbody tr').count();
    expect(visibleRows).toBeGreaterThanOrEqual(5); // 表格展开前可见
  });

  test('1280px - 3:1布局，7列可见', async ({ page }) => {
    await page.setViewportSize({ width: 1280, height: 800 });
    
    await takeScreenshot(page, 'layout-1280px');
    
    const columns = await page.locator('.t-table-header th').allTextContents();
    // 不包含"负责人"和"更新时间"
    expect(columns).not.toContain('负责人');
    expect(columns).not.toContain('更新时间');
  });

  test('操作列 - 图标化，tooltip正常', async ({ page }) => {
    await page.setViewportSize({ width: 1280, height: 800 });
    
    // 操作列仅图标
    const buttons = await page.locator('.action-buttons--compact button').count();
    expect(buttons).toBeGreaterThanOrEqual(3);
    
    // hover第一个按钮
    const firstButton = page.locator('.action-buttons--compact button').first();
    await firstButton.hover();
    
    // 等待tooltip
    await page.waitForSelector('.t-popup-content', { timeout: 3000 });
  });

  test('侧边栏滚动时保持sticky', async ({ page }) => {
    await page.setViewportSize({ width: 1280, height: 600 });
    
    await page.mouse.wheel(0, 500); // 向下滚动
    
    // 侧边栏顶部位置不变
    const sidebarRect = await page.locator('.sidebar').boundingBox();
    expect(sidebarRect?.y).toBeCloseTo(24, 50); // ±50px容差
  });

  test('近期动态可折叠', async ({ page }) => {
    await page.setViewportSize({ width: 1280, height: 1080 });
    
    // 默认收起
    const panel = page.locator('.activity-collapse');
    const timeline = page.locator('.activity-card--timeline');
    await expect(timeline).not.toBeVisible();
    
    // 点击展开
    await panel.locator('.t-collapse-header').click();
    await expect(timeline).toBeVisible();
  });
});
```
